%% Commands for little filled blocks (delta-debug and brute-force)
\def\bffilldark(#1,#2,#3){
  \foreach \x in {#2,...,#3}{
    \node[draw,fill=gt@gray!30,below=\x em of #1] {};
  }
}
\def\bffilllight(#1,#2,#3){
  \foreach \x in {#2,...,#3}{
    \node[draw,fill=gt@cyan!60,below=\x em of #1] {};
  }
}
\def\bffillred(#1,#2,#3){
  \foreach \x in {#2,...,#3}{
    \node[draw,fill=gt@red!60,below=\x em of #1] {};
  }
}
